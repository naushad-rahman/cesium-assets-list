View in Cesium

Using this tileset in Cesium

// In an application that uses CesiumWidget:
var widget = new Cesium.CesiumWidget({
    imageryProvider : new Cesium.TileMapServiceImageryProvider({
        url : '//cesiumjs.org/tilesets/imagery/blackmarble'
    })
});

// In an application that uses Viewer:
var viewer = new Cesium.Viewer({
    imageryProvider : new Cesium.TileMapServiceImageryProvider({
        url : '//cesiumjs.org/tilesets/imagery/blackmarble'
    }),
    baseLayerPicker : false
});

// In other applications:
var scene = ...;
scene.imageryLayers.removeAll(); // optional
scene.imageryLayers.add(new Cesium.TileMapServiceImageryProvider({
    url : '//cesiumjs.org/tilesets/imagery/blackmarble'
}));
